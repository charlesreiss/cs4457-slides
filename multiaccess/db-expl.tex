\begin{frame}{signal strength}
    \begin{itemize}
    \item measure signal stength in dB (decibal)
    \item this is \textit{ratio} and a logarithmic scale
    \item $(X+10)$dB means \textit{ten times power} of $X$dB
    \item $(X+20)$dB means \textit{a hundred times power} of $X$dB
    \end{itemize}
\end{frame}

\begin{frame}{dB and dBm}
\begin{itemize}
\item dB (decimal) is relative unit
\item this may be confusing because you often see sound, wireless quoted in decibels
\vspace{.5cm}
\item for audio: implicitly ratio of sound power to reference
\item for wireless: dBm = dB relative to 1milliwatt (mW) signal
\end{itemize}
\end{frame}
