        \begin{itemize}
        \item reserves (UDP, 10.11.12.13, 12345, any, any)
        \item can have different protocol: (TCP, 10.11.12.13, 12345, any, any)
        \item or different IPs machine has: (UDP, 127.0.0.1, 12345, any, any)
        \end{itemize}

        \end{itemize}
