\begin{frame}{DNSSEC}
    \begin{itemize}
    \item public key infrastructure for DNS
    \item single set of root keys from ICANN/IANA
        \begin{itemize}
        \item not multiple certificate authorities like web PKI
        \end{itemize}
    \item each DNS zone has its own public key
    \item all records within zone have a digital signature
    \item delegation records have keys for subzones
        \begin{itemize}
        \item similar to TLS certificates
        \end{itemize}
    \end{itemize}
\end{frame}


\begin{frame}{DNSSEC offline signing}
    \begin{itemize}
    \item design goal: generate + sign zone
    \item have digital signatures that can verify all records
    \vspace{.5cm}
    \item means: DNS servers don't need to actually sign anything!
    \item can have signing keys offline
        \begin{itemize}
        \item much lower security risk
        \end{itemize}
    \vspace{.5cm}
    \item challenges:
        \begin{itemize}
        \item representing missing domain names (solution sketch: ``no domain names between A and C'' record)
        \end{itemize}
    \end{itemize}
\end{frame}
