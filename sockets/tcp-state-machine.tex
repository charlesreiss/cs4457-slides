\begin{frame}{connection close}
    \begin{itemize}
    \item FIN (I want to close) + ACK (confirm close)
    \vspace{.5cm}
    \item ACK side needs to wait just in case FIN resent
        \begin{itemize}
        \item keep port number from being reused
        \item TIME\_WAIT state
        \end{itemize}
    \end{itemize}
\end{frame}
