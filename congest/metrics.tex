\begin{frame}{congestion control metrics}
    \begin{itemize}
    \item efficiency
        \begin{itemize}
        \item making ``full'' use of available resources
        \end{itemize}
    \item min-max fairness
        \begin{itemize}
        \item everyone gets `equal' resources
        \item often conflict with Pareto-efficiency
        \end{itemize}
    \end{itemize}
\end{frame}
