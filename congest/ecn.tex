\begin{frame}{ECN}
\begin{itemize}
\item explicit congestion notification
\vspace{.5cm}
\item when buffer `close' to full
\item switches/router set `ECN' bit in some packets
    \begin{itemize}
    \item field in IP header
    \end{itemize}
\item ECN bit echoed back in ACKs
\end{itemize}
\end{frame}
